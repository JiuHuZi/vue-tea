### 一、路径上的配置

#### vue.config.js 配置了 `@` 代表 `src`目录

```xml
let path = require('path')
module.exports = {
  configureWebpack: (config) => {
    config.resolve = {
      extensions: ['.js', '.json', '.vue'],
      alias: {
        '@': path.resolve(__dirname, './src')
      }
    }
  }
}

```
